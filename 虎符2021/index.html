# sage
n = 6
a = (4*n ^ 2+12*n-3)
b = 32*(n+3)
ee = EllipticCurve([0, a, 0, b, 0])
# y2=x3+109x2+224x


def orig(P, N):
    x = P[0]
    y = P[1]
    a = (8*(N+3)-x+y)/(2*(N+3)*(4-x))
    b = (8*(N+3)-x-y)/(2*(N+3)*(4-x))
    c = (-4*(N+3)-(N+2)*x)/((N+3)*(4-x))
    da = denominator(a)
    db = denominator(b)
    dc = denominator(c)
    l = lcm(da, lcm(db, dc))
    return [a*l, b*l, c*l]
