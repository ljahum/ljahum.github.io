def check(d, p, n):
    if((p - 1) % n == 0):
        return pow(d, (p - 1) // n, p) == 1
    else:
        k = gcd(n, p - 1)
        return pow(d, (p - 1) // k, p) == 1

def getM(c, e, p):
    for i in range(2, e):
        inv = int(pow(y, i, p))
        tmpc = (c * inverse(inv, p)) % p
        if check(tmpc, p, e):
            return i
    exit(0)
def func(c):
    cq = c % q
    cp = c % p

    m2 = getM(cq, e2, q)
    ic(m1,m2)

    return ans
