from sage.all import continued_fraction, Integer
def wiener(e, n):
    m = 12345
    c = pow(m, e, n)
 
    list1 = continued_fraction(Integer(e)/Integer(n))
    conv = list1.convergents()
    
    for i in conv:
        d = int(i.denominator()) # 分母
        m1 = pow(c, d, n)
        if m1 == m:
            return d
